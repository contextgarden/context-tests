\def\doPageNumber#1%
  {%\the\numexpr #1 - 1\relax\ · #1}% #1 is not a number
   %\the\numexpr \structurelistrealpagenumber - 1\relax\ · #1}
   \tointeger{\structurelistrealpagenumber - 1} · #1}

  [chapter]

\setuplist
  [chapter]
  [pagecommand=\doPageNumber]
