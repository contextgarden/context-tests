\startluacode
    function document.itemize(n)
        if n > 0 then
            context.startitemize()
            for i=1,n do
                context.startitem()
                    context("whatever")
                    if i == 1 then
                        document.itemize(n-1)
                    end
                context.stopitem()
            end
            context.stopitemize()
        end
    end
\stopluacode
